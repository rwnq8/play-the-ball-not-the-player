## Buffer API Execution (Once Credentials Ready)

```python
# File: _buffer_post_playtheball.py
# Usage: python _buffer_post_playtheball.py
# Prerequisites: Buffer token at %USERPROFILE%\.buffer_token_playtheball

import os, urllib.request, json, glob

# --- Load token ---
token_path = os.path.expandvars(r"%USERPROFILE%\.buffer_token_playtheball")
if not os.path.exists(token_path):
    print("[BLOCKED] No Buffer token for Play the Ball. Get token from buffer.com/developers")
    exit(1)
with open(token_path, "r", encoding="utf-8-sig") as f:
    TOKEN = f.read().strip()

# --- GraphQL helper ---
def gql(query, variables=None):
    body = {"query": query}
    if variables: body["variables"] = variables
    data = json.dumps(body).encode("utf-8")
    req = urllib.request.Request("https://api.buffer.com/1/graphql.json", data=data, method="POST")
    req.add_header("Authorization", f"Bearer {TOKEN}")
    req.add_header("Content-Type", "application/json")
    resp = urllib.request.urlopen(req, timeout=15)
    return json.loads(resp.read().decode("utf-8"))

# --- Get org and channels ---
acct = gql("{ account { organizations { id } } }")
org_id = acct["data"]["account"]["organizations"][0]["id"]
channels = gql('{ channels(input: { organizationId: "%s" }) { id service name } }' % org_id)

print("Available channels:")
for ch in channels["data"]["channels"]:
    print(f"  {ch['service']}: {ch['name']} (id={ch['id']})")

# --- Post function ---
def post(channel_id, text, service, link_url=None, now=False):
    post_input = {
        "channelId": channel_id,
        "text": text,
        "schedulingType": "automatic",
        "mode": "shareNow" if now else "addToQueue",
        "assets": [],
    }
    if link_url and service:
        post_input["metadata"] = {service: {"linkAttachment": {"url": link_url}}}
    
    mutation = """mutation($input: CreatePostInput!) { createPost(input: $input) { ... on PostActionSuccess { post { id status text dueAt } } ... on InvalidInputError { message } ... on LimitReachedError { message } } }"""
    
    result = gql(mutation, {"input": post_input})
    post_result = result["data"]["createPost"]
    
    if "post" in post_result:
        return {"ok": True, "id": post_result["post"]["id"], "status": post_result["post"]["status"]}
    return {"ok": False, "error": post_result.get("message", "Unknown")}

# --- Find channels by service ---
twitter_id = next((ch["id"] for ch in channels["data"]["channels"] if ch["service"] == "twitter"), None)
linkedin_id = next((ch["id"] for ch in channels["data"]["channels"] if ch["service"] == "linkedin"), None)
bluesky_id = next((ch["id"] for ch in channels["data"]["channels"] if ch["service"] == "bluesky"), None)

URL = "https://rwnq8.github.io/play-the-ball-not-the-player/"

# --- Post to Twitter/X ---
if twitter_id:
    result = post(twitter_id, 
        "Play the ball, not the player. ⚽\n\nA life lesson from the 2026 World Cup, unpacked across five domains: disagreements, feedback, relationships, career, and leadership.\n\nFull essay: " + URL,
        "twitter", URL, now=True)
    print(f"Twitter: {'OK' if result['ok'] else 'FAIL - ' + result['error']} id={result.get('id')}")

# --- Post to LinkedIn ---
if linkedin_id:
    result = post(linkedin_id,
        "📄 New Essay: Play the Ball, Not the Player\n\nThe 2026 FIFA World Cup is underway. And buried inside every tackle is a principle so clean it deserves to leave the pitch entirely.\n\nRead the full essay + toolkit: " + URL + "\n\n#Leadership #PersonalDevelopment #WorldCup2026",
        "linkedin", URL, now=True)
    print(f"LinkedIn: {'OK' if result['ok'] else 'FAIL - ' + result['error']} id={result.get('id')}")

# --- Post to Bluesky ---
if bluesky_id:
    result = post(bluesky_id,
        "Play the ball, not the player. A life lesson from the 2026 World Cup, unpacked across arguments, feedback, relationships, career, and leadership.\n\n" + URL,
        "bluesky", URL, now=True)
    print(f"Bluesky: {'OK' if result['ok'] else 'FAIL - ' + result['error']} id={result.get('id')}")

print("\nDone. Check buffer.com for scheduled posts.")
```
